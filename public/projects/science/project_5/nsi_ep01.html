def recherche_dicho(tab, cible):
g, d = 0, len(tab) - 1
while g <= d:
m = (g + d) // 2
if tab[m] == cible: return m
elif tab[m] < cible: g = m + 1
else: d = m - 1
return -1

def bfs(graphe, debut):

file = deque([debut])

v = file.popleft()
for w in graphe[v]:

vus.add(w); file.append(w)
